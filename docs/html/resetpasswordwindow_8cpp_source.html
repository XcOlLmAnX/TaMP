150void ResetPasswordWindow::onSetPasswordClicked() {
151 QString p1 = editNewPassword->text().trimmed();
152 QString p2 = editNewPassword2->text().trimmed();
153
154 if (p1.isEmpty() || p2.isEmpty()) {
155 labelStatus3->setText("Заполните оба поля");
156 labelStatus3->setStyleSheet("color: red;");
157 return;
158 }
159 if (p1 != p2) {
160 labelStatus3->setText("Пароли не совпадают");
161 labelStatus3->setStyleSheet("color: red;");
162 return;
163 }
164 if (p1.length() < 6) {
165 labelStatus3->setText("Пароль должен быть не менее 6 символов");
166 labelStatus3->setStyleSheet("color: red;");
167 return;
168 }
169
170 bool hasDigit = false;
171 for (QChar c : p1)
172 if (c.isDigit()) { hasDigit = true; break; }
173
174 if (!hasDigit) {
175 labelStatus3->setText("Пароль должен содержать хотя бы одну цифру");
176 labelStatus3->setStyleSheet("color: red;");
177 return;
178 }
179
180 QString hash = QString::fromStdString(md5(p1.toStdString()));
181 TcpClient::getInstance()->sendMessage(
182 QString("newpass&%1&%2\n").arg(currentLogin, hash));
183}

189void ResetPasswordWindow::onMessageReceived(const QString& message) {
190 if (message.startsWith("reset+")) {
191 QString masked = message.section('&', 1, 1).trimmed();
192 currentLogin = editLogin->text().trimmed();
193 labelMaskedEmail->setText(
194 QString("Код отправлен на почту: %1").arg(masked));
195 labelStatus1->setText("");
196 stack->setCurrentIndex(1);
197 } else if (message.startsWith("reset-")) {
198 labelStatus1->setText("Логин не найден");
199 labelStatus1->setStyleSheet("color: red;");
200 } else if (message.startsWith("confirm+")) {
201 labelStatus2->setText("Код верный!");
202 labelStatus2->setStyleSheet("color: green;");
203 stack->setCurrentIndex(2);
204 } else if (message.startsWith("confirm-")) {
205 labelStatus2->setText("Неверный код");
206 labelStatus2->setStyleSheet("color: red;");
207 } else if (message.startsWith("newpass+")) {
208 labelStatus3->setText("Пароль изменён!");
209 labelStatus3->setStyleSheet("color: green;");
210 emit goToLogin();
211 } else if (message.startsWith("newpass-")) {
212 labelStatus3->setText("Ошибка, попробуйте снова");
