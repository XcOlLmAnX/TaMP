145static bool doSend(const QString& toEmail,
146 const QString& subjectText,
147 const QString& bodyText) {
148 QString senderEmail = SmtpClient::getSender();
149 QString senderPassword = SmtpClient::getSenderPassword();
150
151 QSslSocket socket;
152 socket.connectToHostEncrypted(SmtpClient::smtpHost, SmtpClient::smtpPort);
153 socket.ignoreSslErrors();
154 if (!socket.waitForEncrypted(20000)) {
155 qDebug() << "[SMTP] SSL connection failed:" << socket.errorString();
156 return false;
157 }
158 qDebug() << "[SMTP] SSL connected";
159
160 QString resp = readResponse(&socket, 15000);
161 if (resp.isEmpty() || !resp.startsWith("220")) return false;
162
163 resp = sendCmd(&socket, "EHLO localhost\r\n");
164 if (!resp.startsWith("250")) return false;
165
166 resp = sendCmd(&socket, "AUTH LOGIN\r\n");
167 if (!resp.startsWith("334")) return false;
168
169 resp = sendCmd(&socket, senderEmail.toUtf8().toBase64() + "\r\n");
170 if (!resp.startsWith("334")) return false;
171
172 resp = sendCmd(&socket, senderPassword.toUtf8().toBase64() + "\r\n");
173 if (!resp.startsWith("235")) return false;
174 qDebug() << "[SMTP] Authenticated";
175
176 resp = sendCmd(&socket, "MAIL FROM:<" + senderEmail.toUtf8() + ">\r\n");
177 if (!resp.startsWith("250")) return false;
178
179 resp = sendCmd(&socket, "RCPT TO:<" + toEmail.toUtf8() + ">\r\n");
180 if (!resp.startsWith("250")) return false;
181
182 resp = sendCmd(&socket, "DATA\r\n");
183 if (!resp.startsWith("354")) return false;
184
185 QByteArray subjectB64 = subjectText.toUtf8().toBase64();
186 QByteArray email;
187 email += "Subject: =?UTF-8?B?" + subjectB64 + "?=\r\n";
188 email += "From: " + senderEmail.toUtf8() + "\r\n";
189 email += "To: " + toEmail.toUtf8() + "\r\n";
190 email += "Content-Type: text/plain; charset=UTF-8\r\n";
191 email += "\r\n";
192 email += bodyText.toUtf8() + "\r\n";
193 email += ".\r\n";
194
195 socket.write(email);
196 socket.flush();
197 socket.waitForBytesWritten(5000);
198 resp = readResponse(&socket);
199 if (!resp.startsWith("250")) return false;
200
201 sendCmd(&socket, "QUIT\r\n");
202 socket.disconnectFromHost();
203 if (socket.state() != QAbstractSocket::UnconnectedState)
204 socket.waitForDisconnected(5000);
205
206 qDebug() << "[SMTP] Email sent to" << toEmail;
207 return true;
208}
